CREATE TABLE LUSUARIO (
    ID INT IDENTITY(1,1) PRIMARY KEY,
	CODPERMISSAO INT,
    LOGINUSER VARCHAR(MAX),
    SENHA VARCHAR(MAX),
    EMAIL VARCHAR(MAX),
    NOME VARCHAR(MAX),
	CPF VARCHAR(MAX),
	TELEFONE Varchar(Max),
	DATANASCIMENTO DATE,
	CIDADE VARCHAR(MAX),
	RUA VARCHAR(MAX),
	BAIRRO VARCHAR(MAX),
	CASA VARCHAR(MAX),
	CEP INT
);

CREATE TABLE AVALIACAOALUNO (
    ID INT IDENTITY(1,1) PRIMARY KEY,
    IDALUNO INT,
	DATAAVALIACAO DATE,
	DESCRICAO VARCHAR(MAX),
    BRACODIREITO DECIMAL(18, 2),
	BRACOESQUERDO DECIMAL(18, 2),
	PERNADIREITA DECIMAL(18, 2),
	PERNAESQUERDA DECIMAL(18, 2)
);

CREATE TABLE DINAMICAPERMISSOES (
    ID INT IDENTITY(1,1) PRIMARY KEY,
    DESCRICAO VARCHAR(MAX)
);

CREATE TABLE BOTOESPERMISSOES (
    ID INT IDENTITY(1,1) PRIMARY KEY,
    IDBOTAO INT,
	IDUSUARIO INT
);

CREATE TABLE DINAMICABOTOES (
    ID INT IDENTITY(1,1) PRIMARY KEY,
    DESCRICAO VARCHAR(MAX)
);
